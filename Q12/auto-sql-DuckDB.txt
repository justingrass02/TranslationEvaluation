SELECT O2.ocel_id AS "O2",O3.ocel_id AS "O3",O4.ocel_id AS "O4",O1.ocel_id AS "O1",E1.ocel_id AS "E1",E2.ocel_id AS "E2"
FROM "event_Place Order" AS E1
INNER JOIN event_object AS E2O0 ON E2O0.ocel_event_id = E1.ocel_id
INNER JOIN "object_Orders" AS O1 ON E2O0.ocel_object_id = O1.ocel_id
INNER JOIN object_object AS O2O0 ON O2O0.ocel_source_id = O1.ocel_id
INNER JOIN "object_Items" AS O3 ON O2O0.ocel_target_id = O3.ocel_id
INNER JOIN event_object AS E2O1 ON E2O1.ocel_event_id = E1.ocel_id
INNER JOIN "object_Customers" AS O4 ON E2O1.ocel_object_id = O4.ocel_id
INNER JOIN object_object AS O2O1 ON O2O1.ocel_source_id = O4.ocel_id
INNER JOIN "object_Employees" AS O2 ON O2O1.ocel_target_id = O2.ocel_id
INNER JOIN event_object AS E2O2 ON E2O2.ocel_object_id = O3.ocel_id
INNER JOIN "event_Pick Item" AS E2 ON E2O2.ocel_event_id = E2.ocel_id
WHERE O2.ocel_changed_field IS NULL
AND O3.ocel_changed_field IS NULL
AND O4.ocel_changed_field IS NULL
AND O1.ocel_changed_field IS NULL
