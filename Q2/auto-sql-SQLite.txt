SELECT O1.ocel_id AS "O1",E1.ocel_id AS "E1"
,
CASE WHEN NOT EXISTS (SELECT 1 FROM (SELECT CASE WHEN strftime('%s', E2.ocel_time) - strftime('%s', E1.ocel_time) >= 0 AND strftime('%s', E2.ocel_time) - strftime('%s', E1.ocel_time) <= 864864 THEN 1 ELSE 0 END AS satisfied,
E2.ocel_id AS cnt_key
FROM "event_PayOrder" AS E2
INNER JOIN event_object AS E2O1 ON E2O1.ocel_object_id = O1.ocel_id AND E2O1.ocel_event_id = E2.ocel_id
) AS subqC_0_0_A WHERE subqC_0_0_A.satisfied = 0) THEN 1 ELSE 0 END AS satisfied 
FROM "event_PlaceOrder" AS E1
INNER JOIN event_object AS E2O0 ON E2O0.ocel_event_id = E1.ocel_id
INNER JOIN "object_Orders" AS O1 ON E2O0.ocel_object_id = O1.ocel_id
WHERE O1.ocel_changed_field IS NULL
